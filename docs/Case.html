Example
su.camelCase("bird flight")
// => "birdFlight"
su.camelCase("this_is_a_string")
// => "thisIsAString"
su.camelCase("   _-It is istanbul   not constantinople-_ ", "TR")
// => "itİsİstanbulNotConstantinople"
